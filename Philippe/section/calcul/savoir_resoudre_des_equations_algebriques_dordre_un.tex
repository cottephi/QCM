        	\begin{question}{8}{Calcul}{1}{1215}
				Quelle est la solution de $a.x+b=c$?
            \end{question}
            \begin{reponses}
            	\item[false] $x=c-b-a$
            	\item[false] $x=\frac{c}{a}-b$
                \item[true] $x=\frac{c-b}{a}$
                \item[false] $x=(c+b)/a$
            \end{reponses}

            	\item[false] $x=\frac{4}{3}-2$
