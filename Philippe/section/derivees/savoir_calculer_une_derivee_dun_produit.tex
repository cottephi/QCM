        	\begin{question}{23}{Dérivées}{1}{/}
				Soient $u$ et $v$ deux fonctions dérivables sur $\mathbb{R}$. Quelle est la dérivée de $u\times v$?
            \end{question}
            \begin{reponses}
            	\item[false] $u'v-v'u$
            	\item[false] $u'v'$
                \item[true] $u'v+v'u$
                \item[false] $u'+v'$
            \end{reponses}
