        	\begin{question}{1204}{Vecteurs}{1}{1220}
            	Soient deux vecteurs dans l'espace à 2 dimensions $\vec{A}=(1;1)$ et $\vec{B}=(3;3)$. Quelles sont les coordonnées de $\vec{A}+\vec{B}$?
            \end{question}
            \begin{reponses}
            	\item[false] $(-2;-2)$
            	\item[false] $(2;2)$
                \item[true] $(4;4)$
                \item[false] $(0;0)$
            \end{reponses}
